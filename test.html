$(document).ready(function() { var $modal = $('#imageModal'); var $modalImage = $modal.find('.modal-body img'); var $modalCaption = $modal.find('.modal-caption'); $('.enlarge-image').click(function() { var image = $(this); var caption = image.data('caption'); $modalImage.attr('src', image.attr('src')); $modalCaption.text(caption); $modal.modal('show'); }); });
